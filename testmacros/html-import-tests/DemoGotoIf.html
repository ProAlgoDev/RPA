DemoGotoIf
open /kantu/demo/storeeval
echo How to use gotoIf and label(s) for flow control. For a while/endWhile demo, see the DemoSaveCSV macro.
storeAttribute //input[@id='sometext']@size boxsize
echo input box size =${boxsize}
gotoIf ${boxsize} > 70 BOX-TOO-BIG
type id=sometext This box is ${boxsize} chars wide
storeEval document.title = ${boxsize};
gotoLabel END
label BOX-TOO-BIG
echo Input box too big. This is just a test of gotoIf
storeEval document.title = "Just a gotoIf test. This line should not be reached unless you edit the macro"
label END
echo test case completed
