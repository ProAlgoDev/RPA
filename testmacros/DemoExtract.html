open /kantu/demo/storeeval
echo Current page URL = ${!URL}
echo Current loop value = ${!LOOP}
echo This macro shows various methods to extract and save data from a website
storeAttribute css=img.responsive-img@src mylink
echo href=${mylink}
storeAttribute css=img.responsive-img@alt myalttext
echo alt text = ${myalttext}
storeAttribute //input[@id='sometext']@size boxsize
echo input box size =${boxsize}
type id=sometext This box is ${boxsize} chars wide
storeEval document.title = ${boxsize};
assertTitle 70
click //*[@id="content"]/div[2]/div/h2[3]
storeText //*[@id="content"]/div[2]/div/h2[3] myheader
echo header = ${myheader}
storeTitle mytitle
echo page title = ${mytitle}
echo Last but not least, taking a screenshot is another way to extract data
captureScreenshot myscreenshot-${mytitle}
